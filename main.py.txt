import keyboard

def on_key_press(event):
    if event.name == 'f1':
        print("F1 pressed: Trigger action 1")#tab 1
    elif event.name == 'f2':
        print("F2 pressed: Trigger action 2")#tab 2
    elif event.name == 'f3':
        print("F3 pressed: Trigger action 3")#tab 3
    elif event.name == 'scroll_lock':
        print("Scroll Lock pressed: Trigger action 4")#tab 4
    elif event.name == 'pause':
        print("Pause pressed: Trigger action 5")#home

keyboard.on_press(on_key_press)
keyboard.wait()  # Keep the script running
